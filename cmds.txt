python3 -m uvicorn src.main:app --reload

# --- curl commands ---

# Trigger a build (defaults)
curl -s -X POST http://localhost:8000/build -H "Content-Type: application/json" -d '{}'

# Trigger a build with explicit options
curl -s -X POST http://localhost:8000/build -H "Content-Type: application/json" -d '{"output_dir": "releases"}'

# Trigger a build with config injection
curl -s -X POST http://localhost:8000/build -H "Content-Type: application/json" -d '{"output_dir": "releases", "config": "key: value\n", "config_path": "src/config.yaml"}'

# List all builds
curl -s http://localhost:8000/builds

# Get status of a specific build
curl -s http://localhost:8000/status/<build-id>

# Poll status until done
watch -n 2 "curl -s http://localhost:8000/status/<build-id>"

# Download the zip artifact when build is done
curl -O -J http://localhost:8000/download/<build-id>

# Delete / cleanup a build
curl -s -X DELETE http://localhost:8000/build/<build-id>

# Trigger a build with minimal config (platforms: instagram)
curl -s -X POST http://localhost:8000/build \
  -H "Content-Type: application/json" \
  -d '{"config": "{\"platforms\": \"instagram\"}", "config_path": "port_config.json"}'
